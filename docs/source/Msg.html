/**
 *  Ext.Msg 의 간소화 시킨 함수입니다.
 */
Ext.define('Miraeweb.utils.Msg',{
    alternateClassName : ['Miraeweb.Msg'],
    extend     : 'Ext.Base',
    singleton  : true,

    /**
     *
     * alert창 띄우기
     *
     * @param{String} title  제목
     * @param{String} message 메시지
     * @param{Function} callbackFunc 콜백함수
     */
    alert: function (title, message, callbackFunc) {

        if(Ext.isObject(title)){

            var objMsg =title;
            title = objMsg.title;
            callbackFunc = message;
            message = objMsg.message;

        }

        Ext.Msg.alert(title, message, function (btn) {
            if(Ext.isFunction(callbackFunc)){
                callbackFunc();
            }
        });
    },

    /**
     *
     *  Ext.Msg.confirm의 간소화 버전입니다.
     *
     * @param {String} title  제목
     * @param {String} message 메시지
     * @param {Function} callbackFunc [Yes] 버튼을 클릭시 실행되는 함수
     * @param {Object} scope
     */
    confirm : function(title,message,callbackFunc,scope){
        var msgbox = Ext.Msg.confirm(title,message,function(btn){
            if(Ext.isFunction(callbackFunc)){

                if(btn=='yes'){
                    Ext.callback(callbackFunc,this);
                }
            }else{

                if(btn=='yes'){

                    Ext.callback(callbackFunc.yesFunc,this);

                    callbackFunc.yesFunc();

                }else{

                    Ext.callback(callbackFunc.noFunc,this);
                }
            }

        },scope);
        Ext.WindowMgr.bringToFront(msgbox);
    },

    toast: function (msg, callbackFunc) {

        Ext.widget('toast',{
            html : msg,
        }).show();
        Ext.defer(function(){
            if(Ext.isFunction(callbackFunc)){
                callbackFunc();
            }
        });
    },



});
